    function IsDarkMode()
    {
        // check for explicit user option
        if (localStorage.getItem('darkmode'))
        {
            return localStorage.getItem('darkmode') == 'true';
        }
        else
        {
            return window.matchMedia('(prefers-color-scheme: dark)').matches ? true : false;
        }
    }

    function SetDarkMode(dark, shouldSave)
    {
        document.getElementsByTagName("html")[0].setAttribute("data-bs-theme", dark ? "dark" : "light");

        if (shouldSave)
        {
            localStorage.setItem('darkmode', dark);
        }

        g_IsDarkMode = dark;
    }

    window.onload = function ()
    {
        SetDarkMode(IsDarkMode(), false);
    }

    window.addEventListener('DOMContentLoaded', () =>
    {
        SetDarkMode(IsDarkMode(), false);
    });
</script>
